(Nov 2015) Final working inheritance pattern using Angular 1.4

  • Make further use of the '$' prefixed variable hiding
  • Angular's $controller service is used to instantiate a controller
  • Controller instance is stored in $ctrl in the model
  • wrapObject() to force setting a subInstance property to instead set the parent's property (like an ES6 Proxy setter, but in ES5)
  • subController() does the magic
  • Note: ngModel interface used to follow common model paradigm

myExtend service:

// instance extensions
angular.factory('myExtend', ['$controller', function($controller) {
    return {
        wrapObject: function(obj) {

            // wrap with setters to the prototype
            var props = {};

            for(var key in obj) {

                if (typeof obj[key] != "function") {
                    props[key] = {

                        "get": (function (key) {

                            return function () {

                                return obj[key];
                            };
                        })(key),

                        "set": (function (key) {

                            return function (val) {

                                obj[key] = val;
                            };
                        })(key)
                    }
                }
            }

            return Object.create(obj, props);
        },

        subController: function($scope, name) {

            var model = $scope.$ctrl.ngModel;
            var baseCtrl = model.$ctrl;
            var subCtrl = baseCtrl;

            if (baseCtrl) {
                subCtrl = this.wrapObject(baseCtrl);
            }
            else {
                if (name) {
                    subCtrl = $controller(name, {$scope: $scope});
                }
                else {
                    subCtrl = $scope.$ctrl;
                }
            }

            model.$ctrl = subCtrl;

            $scope.$on('$destroy', function() {
                if (baseCtrl) {
                    model.$ctrl = baseCtrl;
                }
                else {
                    delete model.$ctrl;
                }
            });

            return subCtrl;
        },

        component: {

            setModels: function () {

                if ('modelDef' in this) {

                    // set up models based on definitions
                    this.model = this.model || {};

                    for (var defKey in this.modelDef) {

                        var def = this.modelDef[defKey];
                        var model;

                        if (def.list) {
                            model = [];
                            if ('sortBy' in def) {
                                model.$sort = def.sortBy;
                            }
                            if (def.list.paged) {
                                model.$offset = 0;
                            }
                        }
                        else {
                            model = {};
                        }
                        this.model[defKey] = model;
                    }
                }

                if ('filterDef' in this) {

                    this.filter = this.filter || {};
                    this.filter.$def = this.filterDef;
                }
            }
        }
    };
}]);

myFilter directive+controller, with FilterCtrl base controller:

angular.module('my.directives.filter', ['my.directives.dateFilter', 'my.directives.listFilter'])

    .directive('myFilter', function() {

        return {
            require: ['ngModel'],
            restrict: 'E',
            scope: {},
            templateUrl: '/directives/filter.html',
            controllerAs: '$ctrl',
            bindToController: {
                ngModel: '='
            },

            controller: ['$scope', '$controller', 'myExtend', function($scope, $controller, myExtend) {

                var model = $scope.$ctrl.ngModel;
                var ctrl = myExtend.subController($scope, 'FilterCtrl');
                var $def = model.$def;

                angular.extend(ctrl, {

                    components: {},

                    displayString: function (defKey) {

                        var def = $def.options[defKey];
                        if (defKey in ctrl.components) {
                            return (def.label + ': ' + ctrl.components[defKey].displayString());
                        }
                        return (def.label + ': ' + model[defKey]);
                    },

                    addFilterComponent: function (name, componentCtrl) {

                        ctrl.components[name] = componentCtrl;
                    },

                    defaultOrEmpty: function (defKey) {

                        var def = $def.options[defKey];

                        if (def.type == 'text') {
                            return (model[defKey] == '');
                        }
                        else if (defKey in ctrl.components) {

                            return ctrl.components[defKey].defaultOrEmpty();
                        }

                        return true;
                    },

                    hasFilters: function () {

                        for (var defKey in $def.options) {

                            if (!ctrl.defaultOrEmpty(defKey)) {
                                return true;
                            }
                        }

                        return false;
                    },

                    resetFilter: function (defKey) {

                        var def = $def.options[defKey];

                        if (def.type == 'text') {
                            model[defKey] = '';
                        }
                        else {
                            ctrl.components[defKey].reset();
                        }
                    },
                    resetFilters: function () {

                        for (var defKey in $def.options) {

                            ctrl.resetFilter(defKey);
                        }
                    }
                });

                return ctrl;
            }]
        }
    })

    .controller('FilterCtrl', ['$rootScope', '$scope', '$controller', 'myExtend', function($rootScope, $scope, $controller, myExtend) {

        var model = $scope.$ctrl.ngModel;
        var ctrl = myExtend.subController($scope);
        var $def = model.$def;

        angular.extend(ctrl, {

            customString: function() {
                var paramDef = $def && $def.options;
                if (!paramDef) {
                    return;
                }

                var customArr = [];
                for (var defKey in model) {

                    if (!model.hasOwnProperty(defKey) || (defKey[0] == '$') || (model[defKey] == null)) {
                        continue;
                    }

                    if (defKey in paramDef) {
                        if (paramDef[defKey].type == 'text') {

                            if (model[defKey] != '') {
                                customArr.push("custom string");
                            }
                        }
                        else {
                            var customStr = model[defKey].$ctrl.customString();
                            if (customStr) {
                                customArr.push(customStr);
                            }
                        }
                    }
                }
                return customArr.join(","); // separator
            }
        });

        if (!$def) {
            return ctrl;
        }

        for (var defKey in $def.options) {

            var def = $def.options[defKey];

            if (!(defKey in model)) {

                switch(def.type) {

                    case 'text':
                        model[defKey] = '';
                        break;

                    case 'ListFilterCtrl':
                        model[defKey] = [];
                        break;

                    default:
                        model[defKey] = {};
                }
            }

            if (def.type != 'text') {

                var filterValue = model[defKey];
                if ('$def' in def) {
                    filterValue.$def = def.$def;
                }
                var $ctrl = filterValue.$ctrl;
                if (!$ctrl) {
                    $ctrl = { name: defKey, ngModel: filterValue };
                    $controller(
                        def.type,
                        {$scope: angular.extend($rootScope.$new(true), {$ctrl: $ctrl})}
                    );
                }
            }
        }

        return ctrl;
    }]);

myListFilter directive+controller, and ListFilterCtrl base controller:

angular.module('my.directives.listFilter', [])

    .directive('myListFilter', function() {

        return {
            require: ['^myFilter', 'ngModel'],
            restrict: 'E',
            scope: {},
            templateUrl: '/directives/listFilter.html',
            controllerAs: '$ctrl',
            bindToController: {
                name: '=',
                ngModel: '='
            },

            controller: ['$scope', 'myExtend', '$filter', function($scope, myExtend, $filter) {

                var model = $scope.$ctrl.ngModel;
                var ctrl = myExtend.subController($scope, 'ListFilterCtrl');

                if (!('$def' in model)) { // set up defaults (for testing only)

                    model.$def = {};
                }

                var $def = model.$def;

                if (!('options' in $def)) { // set up defaults (for testing only)

                    $def.options = {
                        "": { index: 0, label: "(test)" }
                    };
                }

                var modelMap = ctrl.modelMap;

                angular.extend(ctrl, {

                    defaultOrEmpty: function () {

                        if (ctrl.isEmpty()) {
                            return true;
                        }

                        var defaultKey = $filter('myOrderObjectBy')($def.options, 'index')[0]._key;

                        for (var key in modelMap) {
                            if ((key != defaultKey) && modelMap[key]) {
                                return false;
                            }
                        }

                        return true;
                    },

                    displayString: function() {

                        var strArr = [];

                        ctrl.forEach(function(key) {

                            strArr.push($def.options[key].label);
                        });

                        return strArr.join(', ');
                    }
                });
            }],

            link: function ($scope, $element, $attrs, ctrls) {

                var ctrl = $scope.$ctrl = $scope.$ctrl.ngModel.$ctrl;

                var filterCtrl = ctrls[0];
                filterCtrl.addFilterComponent(ctrl.name, ctrl);
            }
        }
    })

    .controller('ListFilterCtrl', ['$scope', '$filter', 'myExtend', function($scope, $filter, myExtend) {

        var model = $scope.$ctrl.ngModel;
        var ctrl = myExtend.subController($scope);

        var $def = model.$def;

        var modelMap = {};

        angular.extend(ctrl, {

            modelMap: modelMap,

            isEmpty: function () {

                for (var key in modelMap) {
                    if (modelMap[key]) {
                        return false;
                    }
                }
                return true;
            },

            clear: function() {

                if (!$def || !$def.options) {
                    return;
                }

                for (var key in $def.options) {

                    modelMap[key] = false;
                }
            },

            reset: function () {

                ctrl.clear();

                if (!$def || !$def.options) {
                    modelMap[''] = true;
                }
                else {
                    var defaultKey = $filter('myOrderObjectBy')($def.options, 'index')[0]._key;
                    modelMap[defaultKey] = true;
                }
            },

            customString: function() {

                if (ctrl.isEmpty()) {
                    return;
                }

                var customArr = [];

                ctrl.forEach(function(key) {

                    customArr.push("custom string");
                });

                return customArr.join(","); // separator
            },

            forEach: function(handler) {

                if (ctrl.isEmpty()) {
                    return;
                }

                var optionsArr = $filter('myOrderObjectBy')($def.options, 'index');

                for (var i=0, len=optionsArr.length; i<len; i++) {

                    var key = optionsArr[i]._key;

                    if (key == '') {
                        continue;
                    }

                    if (modelMap[key]) {
                        handler(key);
                    }
                }
            },

            valueString: function() {

                if (ctrl.isEmpty()) {
                    return "";
                }

                var strArr = [];

                ctrl.forEach(function(key) {

                    strArr.push(key);
                });

                return strArr.join(',');
            }
        });

        $scope.$watchCollection("$ctrl.ngModel", function() {

            ctrl.clear();

            model.forEach(function(key) {

                modelMap[key] = true;
            });
        });

        $scope.$watchCollection("$ctrl.modelMap", function() {

            model.length = 0;

            ctrl.forEach(function(key) {

                model.push(key);
            });

            if (!$def) {
                if (model.length == 0) {
                    model.push('');
                }
            }
        });

        if (model.length == 0) {
            ctrl.reset(); // set default value
        }

        return ctrl;
    }]);

myDateFilter directive+controller, and DateFilterCtrl controller:

var labels = {
    'before': 'Before This Date',
    'after': 'After This Date',
    'on': 'On This Date',
    'custom': 'Custom Range'
};

var emptyValue = {
    starttime: 0,
    endtime: 0
};

angular.module('my.directives.dateFilter', ['my.directives.datepicker'])

    .directive('myDateFilter', function() {

        return {
            require: ['^myFilter', 'ngModel'],
            restrict: 'E',
            scope: {},
            templateUrl: '/directives/dateFilter.html',
            controllerAs: '$ctrl',
            bindToController: {
                name: '=',
                ngModel: '='
            },

            controller: ['$scope', 'myExtend', function($scope, myExtend) {

                var model = $scope.$ctrl.ngModel;
                var ctrl = myExtend.subController($scope, 'DateFilterCtrl');

                angular.extend(ctrl, {

                    reset: function () {
                        angular.extend(model, emptyValue);
                    },

                    defaultOrEmpty: function() {
                        return ctrl.isEmpty(); // default value not supported
                    },

                    displayString: function() {

                        if (ctrl.isEmpty()) {
                            return '';
                        }

                        if (ctrl.selectionModel == 'custom') {
                            return (labels[ctrl.selectionModel] + ' : ' + ctrl.date + ' - ' + ctrl.customEndDate);
                        }

                        return (labels[ctrl.selectionModel] + ' : ' + ctrl.date);
                    },

                    options: [
                        {value: 'before', label: labels.before},
                        {value: 'after', label: labels.after},
                        {value: 'on', label: labels.on},
                        {value: 'custom', label: labels.custom}
                    ]
                });

                $scope.$watchCollection(function() {
                    return {
                        date: ctrl.date,
                        customEndDate: ctrl.customEndDate,
                        selectionModel: ctrl.selectionModel
                    };
                }, function(vals, oldVals) {

                    if (angular.equals(vals, oldVals)) {
                        return;
                    }

                    if (!ctrl.date || (ctrl.selectionModel == 'custom' && (!ctrl.customEndDate || (moment(ctrl.date) >= moment(ctrl.customEndDate))))) {
                        ctrl.reset();
                        return;
                    }

                    switch (ctrl.selectionModel) {

                        case 'before':
                            model.starttime = 1;
                            model.endtime = moment(ctrl.date).startOf("day").unix() - 1;
                            break;

                        case 'after':
                            model.starttime = moment(ctrl.date).endOf("day").unix() + 1;
                            model.endtime = moment().endOf("day").unix();
                            break;

                        case 'on':
                            model.starttime = moment(ctrl.date).startOf("day").unix();
                            model.endtime = moment(ctrl.date).endOf("day").unix();
                            break;

                        case 'custom':
                            model.starttime = moment(ctrl.date).startOf("day").unix();
                            model.endtime = moment(ctrl.customEndDate).endOf("day").unix();
                            break;
                    }
                });

                $scope.$watchCollection(function() {
                    return {
                        starttime: model.starttime,
                        endtime: model.endtime
                    };
                }, function() {

                    if (!model.starttime || !model.endtime) {
                        ctrl.date = '';
                        ctrl.customEndDate = '';
                    }
                });
            }],

            link: function ($scope, $element, $attrs, ctrls) {

                var ctrl = $scope.$ctrl = $scope.$ctrl.ngModel.$ctrl;

                var filterCtrl = ctrls[0];
                filterCtrl.addFilterComponent(ctrl.name, ctrl);

                angular.extend(ctrl, {

                    getDatepickerId: function (modelName) {

                        // the jQuery datepicker widget creates dynamic IDs
                        return $element.find("[ng-model='$ctrl." + modelName + "']")[0].id;
                    }
                });
            }
        }
    })

    .controller('DateFilterCtrl', ['$scope', '$filter', 'myExtend', function($scope, $filter, myExtend) {

        var model = $scope.$ctrl.ngModel;
        var ctrl = myExtend.subController($scope);

        if (!('starttime' in model)) { // init

            angular.extend(model, emptyValue);
        }

        angular.extend(ctrl, {

            date: '',
            customEndDate: '',
            selectionModel: 'before',

            isEmpty: function () {
                return ((model.starttime == 0) && (model.endtime == 0));
            },

            queryString: function() {
                if (!ctrl.isEmpty()) {
                    return "((" + ctrl.name + ".starttime eq " + model.starttime + ") and (" + ctrl.name + ".endtime eq " + model.endtime + "))";
                }
            }
        });


        if (ctrl.isEmpty()) {
            return ctrl;
        }

        var dateFilter = $filter('date');

        if (model.starttime == 1) {
            ctrl.selectionModel = 'before';
            ctrl.date = dateFilter((model.endtime+1)*1000, 'MM/dd/yyyy');
        }
        else if (model.starttime % 2) {
            ctrl.selectionModel = 'after';
            ctrl.date = dateFilter((model.starttime-1)*1000, 'MM/dd/yyyy');
        }
        else if ((model.endtime - model.starttime + 1) == 86400) {
            ctrl.selectionModel = 'on';
            ctrl.date = dateFilter(model.starttime*1000, 'MM/dd/yyyy');
        }
        else {
            ctrl.selectionModel = 'custom';
            ctrl.date = dateFilter(model.starttime*1000, 'MM/dd/yyyy');
            ctrl.customEndDate = dateFilter(model.endtime*1000, 'MM/dd/yyyy');
        }

        return ctrl;
    }]);

(Oct 2015) Leveraging the '$' prefixed variable hiding in AngularJS

Did you know?

A "feature" of AngularJS is in hiding variables that have names beginning with '$'. These variables are ignored during watches, and that makes them useful for hiding your own information. Knowing this, one can set up a data model with the definition of the data in a '$def' variable. So running a watch on the data model will work as normal (and ignore changes in the $def variable). This enables easy augmentation of existing simple models to be more complex.

(Dec 2014) Inheritance patterns in JavaScript (and some Python)

SubClassing

  • Traditional
  • Computer Sciencey
  • ES6 'class' keyword
  • used below as "inheritance"

  • Example: Dogs and Cats are Animals. Dogs cannot meow(). Cats cannot bark().

    var Animal = function(){};
Animal.prototype = { walk: function(){} };

var Dog = function(){};
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.bark = function(){};

var Cat = function(){};
Cat.prototype = Object.create(Animal.prototype);
Cat.prototype.meow = function(){};

var dog = new Dog;
var cat = new Cat;

dog instanceof Animal; // true
dog instanceof Dog; // true
dog instanceof Cat; // false

dog.bark; // defined
dog.meow; // not defined

SubInstancing

  • "Dynamic SubClassing"
  • JQuery $.extend()
  • ES6 Object.assign(), copying traits over
  • used below as "mixin"

  • Example: Animals can be Dogs or Cats, given their traits of bark() and meow().

    // from example above
dog.meow = cat.meow;
dog.meow; // Now defined. But conflicts with instanceof operator result of Cat.

// use of mixins
var dogMixin = { bark: function(){} };
var catMixin = { meow: function(){} };

var animal = { walk: function(){} };
var dog = $.extend({}, animal, dogMixin);
var cat = $.extend({}, animal, catMixin);

dog.bark; // defined
dog.meow; // not defined

// The instanceof operator can't be used. Instead, error out or use duck-typing.

JavaScript Inheritance

  • Everything's derived from Object, including Class and Instance
  • Allows for SubInstances in addition to SubClasses
  • Multiple inheritance by mixins, i.e. copy multiple traits over: works due to the two points above

Python Inheritance (I'm not an expert)

  • Class is separate from Instance
  • Does not allow for SubInstances
  • Multiple inheritance (for Class) is built into syntax

When Inheritance breaks down

  • Deep inheritance hierarchies that create rigid dependencies that are very difficult to break when requirements change
  • JavaScript's SubInstance flexibility may alleviate rigidity by providing crutches while the hierarchies are reworked, but this needs to be short-term only
  • In Python, maybe preplan for retiring old things and starting anew

Best Practice

  • For scalability reasons, and the aforementioned breaking down: build shallow and wide, instead of deep and narrow. Favor composition over inheritance to do that.

(Dec 2013) Easy URL Hash Routing with AngularJS

Goals

  • URL Hash Routing Service
  • Drive web application state from the URL

Why?

  • Browsing within a Single Page Application
  • Supporting browser back, forward, refresh buttons
  • Bookmarking and linking from other pages

When?

2013, supporting older Internet Exploder versions

What?

Existing options at the time:

  • ngRoute module
  • External ui-router package
  • Leveraging $location (the winner)
  • Vanilla JavaScript, non-Angular

How?

Supporting older browsers means no HTML5 History API...

Basic URL format:

http://host/path#?param1=value1&param2=value2&paramN=valueN

Examples:

  • ?product=analytics
  • ?product=analytics&view=report
  • ?product=analytics&view=report&sort=date
  • ?product=analytics&view=detail
  • ?product=finance&view=detail
  • ?product=finance&view=detail&filter=abc

Entities:

  • Location model: contains the name/value pairs
  • Root scope: contains the Location model
  • Location service: Built on $location, exposes a register() API
  • Child scopes

Components (directives) that use browser navigation, define URL parameters by using the Location service's register(). All bindings to such parameters by any component are done on the Location model. Example actions:

  • Switch views based on URL parameter 'view': ng-switch on $root.locationModel.view, target 'report' vs 'detail' etc.

    <div ng-switch="$root.locationModel.view">
    <overview ng-switch-when="overview"></overview>

When components are destroyed, any parameters they defined are removed from locationModel automatically. URL parameters with default values, set with register(), are hidden from the URL.

Code:

// used to provide browser history support: two-way binds the location params with a model
angular.module('my.services', []).factory('myLocation', ['$rootScope', '$location', function($rootScope, $location) {

    var locationModel = {};
    $rootScope.locationModel = locationModel;

    var paramDefaults = {};

    var updateLocationModel = function() {

        var locationParams = $location.search();

        // take location params and update the model, setting defaults as needed
        for (var param in paramDefaults) {

            if (param in locationParams) {
                if (typeof paramDefaults[param] == "number") {
                    locationModel[param] = parseInt(locationParams[param]);
                }
                else {
                    locationModel[param] = locationParams[param];
                }
            }
            else {
                locationModel[param] = paramDefaults[param];
            }
        }

        // copy any remaining non-default location params
        for (var param in locationParams) {

            if (!(param in paramDefaults)) {
                locationModel[param] = locationParams[param];
            }
        }
    };

    var updateLocationSearch = function() {

        var replaceLocation = true;
        var oldLocationParams = $location.search();
        var locationParams = {};

        // only include non-default parameters in location.search
        for (var param in paramDefaults) {

            if (locationModel[param] != paramDefaults[param]) {
                locationParams[param] = locationModel[param];

                if (locationParams[param] != oldLocationParams[param]) {
                    replaceLocation = false;
                }
            }
        }

        // also include any remaining params
        for (var param in locationModel) {

            if (!(param in paramDefaults)) {
                locationParams[param] = locationModel[param];

                if (locationParams[param] != oldLocationParams[param]) {
                    replaceLocation = false;
                }
            }
        }

        if (Object.keys(locationParams).length == 0) {
            // prevent an infinite digest (see Angular issues 9635 and 10228)
            // ...remove after Angular upgrade fixes bug.
            return;
        }

        if (replaceLocation) {
            $location.replace();
        }

        $location.search(locationParams);
    };

    var registrations = 0;
    var deregistrationToFunction = null;
    var deregistrationFroFunction = null;

    return {

        register: function($scope, scopeParamDefaults) {

            angular.extend(paramDefaults, scopeParamDefaults);

            updateLocationModel();

            if (registrations == 0) {
                // bind location.search params to model
                deregistrationToFunction = $rootScope.$on('$locationChangeStart', updateLocationModel);

                // bind model to location.search params
                deregistrationFroFunction = $rootScope.$watchCollection("locationModel", updateLocationSearch);
            }

            registrations++;

            $scope.$on("$destroy", function() {

                for (var param in scopeParamDefaults) {
                    delete locationModel[param];
                    delete paramDefaults[param];
                }

                registrations--;

                if (registrations == 0) {
                    deregistrationToFunction();
                    deregistrationFroFunction();
                }
            });
        }
    };
}])

Contracts:

  • Child components should not define the same parameter names as a parent. If needed, use namespaces.
  • Order of parameters is the order of register() invocations, but not guaranteed. This means navigation within an app generally has good order. But if a bookmark link is navigated to, the order of parameters may change upon navigation, though it won't add an extra navigation.

Derived Goals

  • Easy to learn and use, both internally (engineering) and externally (marketing)

Derived Ungoals

  • Internal application (non-URL) routing
  • Parameter collision
  • Complexity

Best Practice

  • Angular 1 allows you to create a standalone controller without a directive. This is especially notable when using the ngRoute module. For scalability reasons, and planning for Angular 2: All controllers should be in directives. This also enables use of the above easy routing and watchers, instead of ngView and ngRoute promises.

(Jan 2013) Fiddle Fingers

History

  • Invented by me in 2002
  • Result of an internal contest at Xilinx, a semiconductor company

Inputs

  • 4 finger-buttons
  • 1 reset
  • 1 on-board 555 clock

Outputs

  • 4 level-LEDs
  • 4 value-LEDs
  • 1 7-segment score display

Description

Fiddle Fingers is designed to be used in child development centers, handicapped rehabilitation centers or simply as a toy for all ages. It is a game that tests finger-eye reflex coordination. The game consists of four buttons that the player places their four fingers on, which excludes the thumb. When the game is started, one LED in four, lights up. The player then depresses the finger that corresponds to the LED. Then, the lit LED turns off and another one lights up. The player has to then depress that LED button and no other finger. This process is repeated and gets faster and faster.

The aim is to be quick at pressing the button corresponding to the LED that is on. This tests the player's eye-finger reflex speed. It is a good tool to improve the coordination of misfortunate disabled people who suffer with respect to hand-eye coordination.

A display of four LEDs shows the player what level they have achieved. There are five LEDs. The game starts with none of the four LEDs lit up, hence level 0. Then 1 lights up as the speed increases, then another and so on till all four are lit up. Once this level is completed the game stops. It should be very difficult if not impossible to get to this stage.

Another display which is a 7-segment display shows the player's current score. This is a count of how many LED presses the player has missed. When this reaches six the game ends. Hence, the best player has the least score, and the highest level.

Possible future modifications are increasing the score limit, cycling through the levels to attain more levels, changing the initial start speed of the game etc.
